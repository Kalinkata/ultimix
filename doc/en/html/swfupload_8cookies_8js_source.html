22  SWFUpload.prototype.refreshCookies = function (sendToFlash) {
23  if (sendToFlash === undefined) {
24  sendToFlash = true;
25  }

33  for (i = 0; i < caLength; i++) {

35 
36  // Left Trim spaces

46  }
